171  class string_comparison_allowed
172  {
173  constexpr static bool l_is_vstring_exact = std::is_same<VString<sizeof(L) - 1>, L>::value;
174  constexpr static bool l_is_vstring_approx = std::is_base_of<VString<sizeof(L) - 1>, L>::value;
175  constexpr static bool r_is_vstring_exact = std::is_same<VString<sizeof(R) - 1>, R>::value;
176  constexpr static bool r_is_vstring_approx = std::is_base_of<VString<sizeof(R) - 1>, R>::value;
177 
178  constexpr static bool l_is_restricted = l_is_vstring_approx && !l_is_vstring_exact;
179  constexpr static bool r_is_restricted = r_is_vstring_approx && !r_is_vstring_exact;
180  public:
181  constexpr static bool value = std::is_same<L, R>::value || (!l_is_restricted && !r_is_restricted);
182  };
