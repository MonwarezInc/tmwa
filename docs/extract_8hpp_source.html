44 template<class T, typename=typename std::enable_if<std::is_integral<T>::value && !std::is_same<T, char>::value && !std::is_same<T, bool>::value>::type>
45 bool impl_extract(XString str, T *iv)
46 {
47  if (!str || str.size() > 20)
48  return false;
49  if (!((str.front() == '-' && std::is_signed<T>::value)
50  || ('0' <= str.front() && str.front() <= '9')))
51  return false;
52  // needs a NUL, but can't always be given one. TODO VString?
53  char buf[20 + 1];
54  std::copy(str.begin(), str.end(), buf);
55  buf[str.size()] = '\0';
56 
57  char *end;
58  errno = 0;
59  if (std::is_signed<T>::value)
60  {
61  long long v = strtoll(buf, &end, 10);
62  if (errno || *end)
63  return false;
64  *iv = v;
65  return *iv == v;
66  }
67  else
68  {
69  unsigned long long v = strtoull(buf, &end, 10);
70  if (errno || *end)
71  return false;
72  *iv = v;
73  return *iv == v;
74  }
75 }

171 template<char split, int n, class F, class... R>
172 bool impl_extract(XString str, Record<split, n, F, R...> rec)
173 {
174  XString::iterator s = std::find(str.begin(), str.end(), split);
175  XString::iterator s2 = s;
176  if (s2 != str.end())
177  ++s2;
178  XString head = str.xislice_h(s);
179  XString tail = str.xislice_t(s2);
180  if (s == str.end())
181  return (extract(head, rec.frist) && n <= 1)
182  || (!head && n <= 0);
183 
184  return (extract(head, rec.frist) || n <= 0)
185  && extract(tail, rec.rest);
186 }
